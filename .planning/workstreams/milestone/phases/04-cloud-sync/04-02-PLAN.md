<task type="auto">
  <name>Task 2: Create sync status store</name>
  <files>src/lib/stores/sync.ts</files>
  <action>
    Create src/lib/stores/sync.ts for managing sync state:
    
    ```typescript
    import { writable, derived } from 'svelte/store';
    
    export type SyncStatus = 'synced' | 'pending' | 'error' | 'syncing';
    
    // Online/offline status
    export const isOnline = writable(true);
    
    // Sync status (pending = local changes not yet synced)
    export const syncStatus = writable<SyncStatus>('synced');
    
    // Last sync timestamp
    export const lastSyncTime = writable<Date | null>(null);
    
    // Has unsynced changes (derived)
    export const hasUnsyncedChanges = derived(
      syncStatus,
      ($syncStatus) => $syncStatus === 'pending' || $syncStatus === 'syncing'
    );
    
    // Initialize online/offline listeners
    if (typeof window !== 'undefined') {
      isOnline.set(navigator.onLine);
      
      window.addEventListener('online', () => {
        isOnline.set(true);
        // Auto-sync when coming back online
        triggerSync();
      });
      
      window.addEventListener('offline', () => {
        isOnline.set(false);
      });
    }
    
    // Manual sync trigger (placeholder - Convex handles automatically)
    export function triggerSync() {
      if (!$isOnline) {
        console.log('Cannot sync while offline');
        return;
      }
      syncStatus.set('syncing');
      // Convex automatically syncs, this is for UI feedback
      setTimeout(() => {
        syncStatus.set('synced');
        lastSyncTime.set(new Date());
      }, 1000);
    }
    ```
  </action>

      >
        {#if !$isOnline}
          <span class="icon">📴</span>
          <span class="message">You're offline. Changes will sync when connected.</span>
        {:else if $syncStatus === 'syncing'}
          <span class="icon">🔄</span>
          <span class="message">Syncing...</span>
        {:else if $syncStatus === 'pending'}
          <span class="icon">⚠️</span>
          <span class="message">Changes pending sync</span>
          <button class="sync-btn" onclick={triggerSync}>Sync now</button>
