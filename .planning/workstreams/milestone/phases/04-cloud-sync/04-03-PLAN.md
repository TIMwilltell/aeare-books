  <name>Task 1: Create JSON export function</name>
  <files>src/lib/api/export.ts</files>
  <action>
    Create src/lib/api/export.ts with export functionality:
    
    ```typescript
    import { db, getAllBooks, getProgressEventsByBook, type Book, type ProgressEvent } from '$lib/db';
    
    export interface ExportData {
      exportedAt: string;
      version: '1.0';
      books: Book[];
      progressEvents: ProgressEvent[];
    }
    
    export async function exportLibrary(): Promise<void> {
      // Get all books from IndexedDB
      const books = await getAllBooks();
      
      // Get all progress events (need to collect from each book)
      const allProgressEvents: ProgressEvent[] = [];
      for (const book of books) {
        if (book.id) {
          const events = await getProgressEventsByBook(book.id);
          allProgressEvents.push(...events);
        }
      }
      
      // Create export data structure
      const exportData: ExportData = {
        exportedAt: new Date().toISOString(),
        version: '1.0',
        books: books.map(b => ({
          ...b,
          // Convert dates to ISO strings for JSON serialization
          readDate: b.readDate ? b.readDate.toISOString() : null,
          quizDate: b.quizDate ? b.quizDate.toISOString() : null,
          createdAt: b.createdAt.toISOString(),
          updatedAt: b.updatedAt.toISOString(),
        })),
        progressEvents: allProgressEvents.map(e => ({
          ...e,
          eventDate: e.eventDate.toISOString(),
          createdAt: e.createdAt.toISOString(),
        })),
      };
      
      // Create and download JSON file
      const jsonString = JSON.stringify(exportData, null, 2);
      const blob = new Blob([jsonString], { type: 'application/json' });
      const url = URL.createObjectURL(blob);
      
      const link = document.createElement('a');
      link.href = url;
      link.download = `aeare-library-${new Date().toISOString().split('T')[0]}.json`;
      document.body.appendChild(link);
      link.click();
      document.body.removeChild(link);
      URL.revokeObjectURL(url);
    }
    ```
